  public int no=0;
  public void setBookCapacity(int value){
    this.capacity = value;
    this.bookname=new String [capacity];
  }
  public void addBook(String name){
    total++;
    if(total<=capacity){
      while(no<total){
      this.bookname[no]=name;
       System.out.println("Book '"+bookname[no]+ "' added to the library");
       no++;
      }
  }
    else{
      System.out.println("Exceeds maximum capacity. You can't add more than " + capacity + " books");
    total--;
    }
  }
  public void printDetail(){
     System.out.println("Maximum Capacity: "+capacity);
     System.out.println("Total Books: " +total);
     System.out.println("Book List:");
     for(int i=0;i<capacity;i++){
       if(bookname[i]==null){
         break;}
       else{
     System.out.println(bookname[i]);
     }
     }
  }
}

public class Cart {

    public double [] price =new double[3];
    public double discountp;
    public double totalp;
    public int itemCount;
    public void create_cart(int no) {
        this.cartno=no;
    }
    public void addItem(String item,double price){
      itemCount++;
      if(itemCount>3){
        System.out.println("You already have 3 items on your cart");
     return; }
      else{
        for(int i=0;i<3;i++){
          if(this.item[i]==null){
          this.item[i]=item;
          this.price[i]=price;
          System.out.println(this.item[i] + " added to cart " + cartno);
          System.out.println("You have " + (i+1) + " item(s) in your cart now");
        break;
          }
      }
    }
}
    public void addItem(double price,String item){
      itemCount++;

     return; }
      else{
        for(int i=0; i<3;i++){
          if(this.item[i]==null){
          this.item[i]=item;
          this.price[i]=price;
          System.out.println(this.item[i] + " added to cart " + cartno);
          System.out.println("You have " + (i+1) + " item(s) in your cart now");
        break;
          }
      }
    }
    }
    public void cartDetails(){
      totalp=0;
      System.out.println("Your cart(c" +cartno+ ")");
        for(int i=0; i<3; i++) {
            if (this.item[i]!=null) {
            System.out.println(this.item[i] + " - " +this.price[i]);
            }
        }
        System.out.println("Discount Applied: "+ discountp +"%");
        for(int i=0;i<3;i++) {

                totalp+=this.price[i];
            }
        }

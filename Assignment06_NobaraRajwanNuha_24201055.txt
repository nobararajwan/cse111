  public static double total_revenue;
  public String name;
  public int discount;
  public double price=150;
  public Book(String nm,int dis){
    name=nm;
    discount=dis;
    if(discount>0.0){
      price= discountedPrice(discount);
    }
    Book.total_books_sold+=1;
    Book.total_revenue+=price;
  }
  public double discountedPrice(double x){
    double j = 150-((x/100.00)*150);
    return j;
  }
  public void bookDetails(){
    System.out.println("Title: "+name+"\nPrice after Discount: "+price+" TK");
  }
}

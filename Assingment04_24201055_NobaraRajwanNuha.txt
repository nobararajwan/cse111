   }
   public void storeID(int id){
     this.id=id;
   }
   public void addCourse(String course){
     if(this.cg==0.0){
       System.out.println("Failed to add "+course+"\nSet CG first");
       return;
     }
     if(this.cg<3.0){
     if(count<3){
       this.course[count]=course;
       count++;
     }
     else{
       System.out.println("Failed to add "+course);
        System.out.println("CG is low. Can't add more than 3 courses.");
     }
     }
     else if(this.cg>=3.0){
       if(count<4){
         this.course[count]=course;
         count++;
       }
       else{
       System.out.println("Failed to add "+course);
        System.out.println("Maximum 4 courses allowed.");
     }
     }
   }
   public void storeCG(double cg){
     this.cg=cg; 
   }
   public void removeAllCourse(){
     for(int i=0;i<count;i++){
       this.course[i]=null;
     }
     this.count=0;
   }
   public void addCourse(String [] courses){
     for(int i=0;i<courses.length;i++){
       addCourse(courses[i]);
     }
   }
   
   public void showAdvisee(){
     System.out.println("Student ID: "+this.id+", "+"CGPA: "+this.cg);
     if(count==0){
       System.out.println("No courses added");
     }
     else if(count>0){
     System.out.println("Added courses area:");
     for(int i=0;i<count;i++){
       if(this.course!=null){
       System.out.print(this.course[i]+" ");
     }
     }
     System.out.println();
     }

class Triangle{
  int length1;
  int length2;
  int length3;
  int perim;
  int count;
  int [] peri=new int[4];
  public Triangle(int a,int b,int c){
    length1=a;
      length2=b;
      length3=c;
      this.perim =a+b+c;
  }
  public void triangleDetails(){
      System.out.println("Three sides of the triangle are: "+length1+","+length2+","+length3);
      System.out.println("Perimeter: "+this.perim);
  }
  public String printTriangleType(){
        if(length1==length2&&length2==length3){
          return "This is an Equilateral Triangle.";
        }
        else if(length1==length2&&length2!=length3||length1!=length2&&length2==length3){
          return "This is an Isosceles Triangle.";
        }
        else{
          return "This is an Scalene Triangle.";
        }
  
  }
  public void compareTrinagles(Triangle x){
    if(this==x){
      System.out.println("These two triangle objects have the same address.");
    }
    else if(this!=x){
      if(this.length1==x.length1&&this.length2==x.length2&&this.length3==x.length3){
        System.out.println("Addresses are different but the sides of the triangles are equal.");
      }
      else if(this.perim==x.perim){
        System.out.println("Only the perimeter of both triangles is equal.");
      }
      else {
        System.out.println("Addresses, length of the sides and perimeter all are different.");
      }
    }
  }
}

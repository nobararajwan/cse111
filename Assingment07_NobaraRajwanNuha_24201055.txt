public class BrowserHistory {
  public String[] pages;
  public int count;
  public  BrowserHistory(int n){
    pages=new String[n];
  }
  public void showHistory(){
    if(count==0){
      System.out.println("No web pages visited yet.");
    }
    else{
      System.out.println("Browser History:");
      for(int i=0;i<count;i++){
        System.out.println(pages[i]);
      }
    }
  }
  public void currentWeb(){
    if(count==0){
      System.out.println("You have not visited any website yet!");
    }
    else{
      System.out.print("Current web page: ");
      System.out.println(pages[count-1]);
    }
  }
  public void visitWeb(String web){
    if(count<3){
      pages[count]=web;
      count++;
      System.out.println("Visited: "+web);
    }
    else{
      System.out.println("History is full. Cannot visit ");
      System.out.println(web+"web page.");
    }
  }
  public void back(){
    count--;
  }
}

    if (customer[0]==null){
      System.out.println("No customers to serve");
    }
    else{
      count--;
      System.out.println("Serving "+customer[0]);
     customer[0]=null;
     int x=1;
     while(x<customer.length){
       
       customer[x-1]=customer[x];
       customer[x]=null;
       x++;
    }
      }
    }
  public void addCustomer(String name){
    if(count<customer.length){
      customer[count]=name;
      count++;
      System.out.println("Added: "+ name);
    }
    else{
      System.out.println("Queue is full. Cannot add "+name);
    }
  }
}

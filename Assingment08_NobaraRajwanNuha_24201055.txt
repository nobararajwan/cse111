   this.sal=sal;
	bonus=b;
  }
  public void calculateSalary(){
 	hours =getHoursWorked();
 	if(hours>40&&hours<80){
   	bonussal=getBaseSalary()+((bonus/100.0)*getBaseSalary());
 	}
  }
  public void requestIncrement(int i){
 	hours =getHoursWorked();
	if(hours>80&&hours<=100){
   	double incre =i/2.0;
  	this.sal+=incre;
  	setBaseSalary(this.sal);
  	System.out.println(incre+" Increment approved.");
	}
	else if(hours>100){
  	this.sal+=i;
  	setBaseSalary(this.sal);
	}
	else {
  	System.out.println("Increment denied.");
	}
  }
  public void displayInfo(){
super.displayInfo();
  System.out.println("Bonus: "+bonus+"%"+"\nFinal Salary: "+"$"+bonussal);
}
  }
